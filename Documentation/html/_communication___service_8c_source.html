191void CommunicationService_Callback(signed char c, int UART_nb)
192{
193 int n =c;
194 if(n > 0 & n <= 100){ // entre 0 et 100
195 if(functionTableDirection[UART_nb-1]){
196 if(functionTableDirection[UART_nb-1][BABORD]){
197 functionTableDirection[UART_nb-1][BABORD](c);
198 }
199 }
200 }
201 else if(c >=0xFFFFFF9C & c <=0xFFFFFFFF){ // entre 0 et 100
202 if(functionTableDirection[UART_nb-1]){
203 if(functionTableDirection[UART_nb-1][TRIBORD]){
204 functionTableDirection[UART_nb-1][TRIBORD](256-c);
205 }
206 }
207 }
208 else if( c >= 0x0 & c <= 0x7F){ // compris entre le char(0) et le char(128)
209 if(functionTable[UART_nb-1]){
210 if(functionTable[UART_nb-1][(int)c]){
211 functionTable[UART_nb-1][(int)c]();
212 }
213 }
214 else if(functionTableBytes[UART_nb-1]){
215 functionTableBytes[UART_nb-1](c);
216 }
217 } // caractère non ASCII
218 else{
219 if(c >= -100 & c <= -1){ // compris entre -100 et -1
220 if(functionTableDirection[UART_nb-1]){
221 if(functionTableDirection[UART_nb-1][BABORD]){
222 functionTableDirection[UART_nb-1][BABORD](c*(-1));
223 }
224 }
225 }else{ //ce qui est < à -100
226 if(functionTableBytes[UART_nb-1]){
227 functionTableBytes[UART_nb-1](c);
228 }
229 }
230 }
231}
